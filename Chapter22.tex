\section{Condensations and Alpha Substitutions of Carbonyl Componds}

\subsection{Introduction 20260528}

\begin{enumerate}
    \item \textbf{Alpha} ($\boldsymbol{\alpha}$) \textbf{substitutions} replace a hydrogen atom at the $\boldsymbol{\alpha}$ \textbf{carbon atom}  by some other group.
    \item Alpha substitutions generally take place when the carbonyl compound is converted to its enolate ion or its enol tautomer.
\end{enumerate}

\begin{mechanism}
    Alpha Substitution
    \begin{itemize}
        \item \textbf{\textit{Step 1:}} Deprotonation of a carbon to form an enolate.
        \item \textbf{\textit{Step 2:}} Nucleophilic attack on an electrophile.
        \begin{figure}[H]
            \centering
            \includegraphics[width=0.7\textwidth]{22.1_Alpha_Substitution.png}
            %\caption{***}
            \label{22.1_Alpha_Substitution}
        \end{figure}
    \end{itemize}
\end{mechanism}

\begin{enumerate}[resume]
    \item Carbonyl \textbf{condensations}: alpha substitutions where the electrophile is another carbonyl compound.
    \item If the electrophile is a ketone or an aldehyde: nucleophilic addition.
\end{enumerate}

\begin{mechanism}
    Addition of an Enolate to Ketones and Aldehydes (a Condensation)
    \begin{itemize}
        \item \textbf{\textit{Step 1:}} The enolate adds to the carbonyl group.
        \item \textbf{\textit{Step 2:}} Protonation of the alkoxide.
        \begin{figure}[H]

    \end{itemize}
\end{mechanism}

\begin{enumerate}[resume]
    \item If the electrophile is an ester: nucleophilic acyl substitution.
\end{enumerate}

\begin{mechanism}
    Substitution of an Enolate on an Ester (a Condensation)
    \begin{itemize}
        \item \textbf{\textit{Step 1:}} Addition of the enolate.
        \item \textbf{\textit{Step 2:}} Elimination of the alkoxide.

\subsection{Enols and Enolate Ions 20260602}

\subsubsection{Keto–Enol Tautomerism}

\begin{enumerate}
    \item A proton on the $\alpha$ carbon atom is abstracted to form a resonance-stabilized enolate ion with the negative charge spread over a carbon atom and an oxygen atom.
    \item Reprotonation can occur either on the $\alpha$ carbon (returning to the \textbf{keto} form) or on the oxygen atom, giving a vinyl alcohol, the \textbf{enol} form.
\end{enumerate}

\begin{mechanism}
    Base-Catalyzed Keto–Enol Tautomerism
    \begin{itemize}
        \item \textbf{\textit{Step 1:}} Deprotonation of a C.
        \item \textbf{\textit{Step 2:}} Reprotonation on O.
        \begin{figure}[H]
            \centering
            \includegraphics[width=0.9\textwidth]{22.2_Keto_Enol_ Tautomerism_B.png}
            %\caption{***}
            \label{22.2_Keto_Enol_ Tautomerism_B}
        \end{figure}
    \end{itemize}
\end{mechanism}

\begin{enumerate}[resume]
    \item This type of isomerization, occurring by the migration of a proton and the movement of a double bond, is called \textbf{tautomerism}, and the isomers that interconvert are called \textbf{tautomers}.
    \item In acid, a proton is moved from the a carbon to oxygen by first protonating oxygen and then removing a proton from carbon.
\end{enumerate}

\begin{mechanism}
    Acid-Catalyzed Keto–Enol Tautomerism
    \begin{itemize}
        \item \textbf{\textit{Step 1}}: An acid protonates the carbonyl oxygen.
        \item \textbf{\textit{Step 2}}: Deprotonation on the a carbon gives the enol form.

    \item If an asymmetric carbon atom has an enolizable hydrogen atom, a trace of acid or base allows that carbon to invert its configuration, with the enol serving as the intermediate. A racemic mixture (or an equilibrium mixture of diastereomers) is the result.

\subsection{Formation and Alkylation of Enamines 20260604}

\begin{enumerate}
    \item An \textbf{enamine} (a \textbf{vinyl amine}) is the nitrogen analogue of an enol.
    \item The nucleophilic carbon atom attacks an electrophile to give a resonance-stabilized cationic intermediate (an iminium ion). % pic
    \item An enamine results from the reaction of a ketone or aldehyde with a \textit{secondary} amine.
    \item Enamines are intermediate in reactivity: more reactive than an enol, but less reactive than an enolate ion.
    \item The enamine alkylation procedure is sometimes called the \textbf{Stork reaction}. The Stork reaction can alkylate or acylate the position of a ketone, using a variety of reactive alkyl and acyl halides.
\end{enumerate}

\subsection{Alpha Halogenation of Ketones 20260602}

\subsubsection{Base-Promoted Alpha Halogenation}

\begin{enumerate}
    \item When a ketone is treated with a halogen and base, an $\alpha$-halogenation reaction occurs. %pic
\end{enumerate}

\begin{mechanism}
    Base-Promoted Halogenation
    \begin{itemize}
        \item \textbf{\textit{Step 1:}} Deprotonation of the a carbon forms the enolate ion.
        \item \textbf{\textit{Step 2:}} The enolate ion attacks the electrophilic halogen.
    \end{itemize}
\end{mechanism}

\begin{enumerate}[resume]
    \item Multiple Halogenation
    \begin{enumerate}
        \item In many cases, base-promoted halogenation does not stop with replacement of just one hydrogen. The product is more reactive toward further halogenation than is the starting material, because the electronwithdrawing halogen stabilizes the enolate ion, enhancing its formation.
    \end{enumerate}
\end{enumerate}

\subsubsection{The Haloform Reaction}

\begin{enumerate}
    \item Methyl ketones have three $\alpha$ protons on the methyl carbon, and they undergo halogenation three times to give trihalomethyl ketones.
\end{enumerate}

\begin{mechanism}
    Final Steps of the Haloform Reaction

    The conclusion of the haloform reaction is a nucleophilic acyl substitution, with hydroxide as the nucleophile and \ch{^-CX3} as the leaving group.
\end{mechanism}

\begin{enumerate}[resume]
    \item When the halogen is iodine, the haloform product (iodoform) is a solid that separates out as a yellow precipitate. This \textbf{iodoform test} identifies methyl ketones, which halogenate three times, then lose to give iodoform.
    \item Iodine is an oxidizing agent, and an alcohol can give a positive iodoform test if it oxidizes to a methyl ketone.
\end{enumerate}

\end{enumerate}

\begin{mechanism}
    Acid-Catalyzed Alpha Halogenation

    Acid-catalyzed halogenation results when the enol form of the carbonyl compound serves as a nucleophile to attack the halogen (a strong electrophile). Deprotonation gives the $\alpha$-haloketone.

    \end{itemize}

\subsection{The Aldol Condensation of  Ketones and  Aldehydes 20260604}

\begin{enumerate}
    \item The aldol condensation involves the nucleophilic addition of an enolate ion to another carbonyl group.
    \item The product, a ketone or aldehyde, is called an aldol because it contains both an aldehyde group and the hydroxy group of an alcohol. The aldol product may dehydrate to an a,b-unsaturated carbonyl compound (ususally the final product).
